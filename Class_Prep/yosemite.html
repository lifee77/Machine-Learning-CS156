from sklearn import linear_model

directory = "data/yosemite-temperatures/yosemite_village"
years = range(2011, 2017)
files = [os.path.join(directory, 'CRNS0101-05-%d-CA_Yosemite_Village_12_W.txt' % y) for y in years]

usecols = [1, 2, 8]

data = [np.loadtxt(f, usecols=usecols) for f in files]
data = np.vstack(data)

data[:, 1] = np.floor_divide(data[:, 1], 100) * 60 + np.mod(data[:, 1], 100)
valid = data[:, 2] > -1000

x_train = data[valid, 1].reshape(-1, 1)
y_train = data[valid, 2]
regr = linear_model.LinearRegression()

x_predict = [[0], [1435]]

plt.scatter(data[valid, 1], data[valid, 2], label='Data', color='b')

plt.legend(loc='best')
plt.show()

from sklearn.metrics.pairwise import rbf_kernel
from sklearn.linear_model import Ridge
import matplotlib.pyplot as plt
import numpy as np

# Try to predict weight as a function of height.
# Physics would suggest that weight is proportional to volume,
# and volume is proportional to height^3.  However we will
# use a linear parameter model with radial basis functions
# to model this nonlinear relationship.

heights = np.asarray([1.32, 1.45, 1.66, 1.88]).reshape(-1, 1)
weights = np.asarray([55, 67, 70, 100]).reshape(-1, 1)
centers = np.asarray([1.0, 1.2, 1.4, 1.6, 1.8, 2.0]).reshape(-1, 1)
print("centers.shape", centers.shape)
print("heights.shape", heights.shape)
print("weights.shape", weights.shape)

# I recommend playing around with these parameters and seeing how well you fit
# the data.  Changing one of the parameters will change the optimal value of
# the other:
sigma = 0.5
alpha = 0.0001

# New representation:
x = rbf_kernel(heights, centers, gamma=1.0 / sigma)
print("x.shape", x.shape)
print("Original inputs", heights)
print("RBF repr", x)

# Now go and fit a linear model between x and height
regr = Ridge(alpha=alpha, fit_intercept=False)
regr.fit(x, weights)
print("Score on training data = ", regr.score(x, weights))
all_x = np.linspace(-3.0, 5.0, 1000).reshape(-1, 1)

# New representation:
expanded_x = rbf_kernel(all_x, centers, gamma=1.0 / sigma)
all_y = regr.predict(expanded_x)
print("all_x.shape", all_x.shape)
print("expanded_x.shape", expanded_x.shape)
print("all_y.shape", all_y.shape)

# Show that the predictions tend to zero far away from inputs
plt.figure()
plt.plot(all_x, all_y)
plt.scatter(heights, weights)
plt.xlabel("Height (m)")
plt.ylabel("Weight (kg)")


# Zoom in and see how well predictions fit the data
zoom_ind = (all_x > heights.min()) & (all_x < heights.max())
plt.figure()
plt.plot(all_x[zoom_ind], all_y[zoom_ind])
plt.scatter(heights, weights)
plt.xlabel("Height (m)")
plt.ylabel("Weight (kg)")
plt.show()

import numpy as np
import matplotlib.pyplot as plt
from scipy.optimize import curve_fit
from sklearn.metrics import mean_squared_error, r2_score
from sklearn.model_selection import train_test_split

# --- Assume x and y are defined from your earlier data loading ---
# x: time in minutes since midnight (shape: (n, 1))
# y: air temperature in Celsius (shape: (n,))

# Split data into training and test sets (80/20 split)
x_train, x_test, y_train, y_test = train_test_split(x, y, test_size=0.2, random_state=42)

# Flatten x_train and x_test to 1D arrays for curve_fit
x_train_flat = x_train.ravel()
x_test_flat = x_test.ravel()

# --- Optional: Subsample training data if the dataset is very large ---
max_samples = 5000  # Adjust this threshold as needed
if x_train_flat.size > max_samples:
    idx = np.random.choice(x_train_flat.size, size=max_samples, replace=False)
    x_train_fit = x_train_flat[idx]
    y_train_fit = y_train[idx]
else:
    x_train_fit = x_train_flat
    y_train_fit = y_train

# Define a sine model for the diurnal temperature cycle
def sine_model(t, A, phi, offset):
    # t: time in minutes since midnight (should be 1D)
    # A: amplitude, phi: phase shift, offset: vertical shift (average temperature)
    return A * np.sin(2 * np.pi / 1440 * t + phi) + offset

# Initial guess for parameters: amplitude=10, phase shift=0, offset=mean temperature
p0 = [10, 0, np.mean(y_train_fit)]

# Fit the sine model using curve_fit on the (possibly subsampled) training data
params, cov = curve_fit(sine_model, x_train_fit, y_train_fit, p0=p0)
A, phi, offset = params
print("Fitted parameters:")
print("Amplitude:", A)
print("Phase shift:", phi)
print("Offset:", offset)

# Predict on the test set using x_test_flat
y_pred = sine_model(x_test_flat, *params)
mse = mean_squared_error(y_test, y_pred)
rmse = np.sqrt(mse)
r2 = r2_score(y_test, y_pred)

print("\nPerformance Metrics on Test Set:")
print("Mean Squared Error (MSE):", mse)
print("Root Mean Squared Error (RMSE):", rmse)
print("R-squared (R²):", r2)

# Visualize predictions vs. true values on the test set
plt.figure(figsize=(8, 6))
plt.scatter(x_test_flat, y_test, color='blue', alpha=0.6, label='True Values')
plt.scatter(x_test_flat, y_pred, color='red', alpha=0.6, label='Sine Model Predictions')
plt.xlabel("Time (minutes since midnight)")
plt.ylabel("Air Temperature (°C)")
plt.title("Non-linear Sine Model Performance on Test Set")
plt.legend()
plt.show()

# Plot the full fitted sine curve over one day along with the data
t_full = np.linspace(0, 1440, 1000)

import numpy as np
import matplotlib.pyplot as plt
from sklearn.kernel_approximation import RBFSampler
from sklearn.linear_model import Ridge, Lasso
from sklearn.pipeline import Pipeline
from sklearn.model_selection import RandomizedSearchCV, train_test_split
from sklearn.metrics import mean_squared_error, r2_score

# ---------------------------
# Split data into training and test sets (80/20 split)
# ---------------------------
x_train, x_test, y_train, y_test = train_test_split(
    x, y, test_size=0.2, random_state=42
)

# ---------------------------
# Optionally subsample training data if it's very large to speed up hyperparameter search
# ---------------------------
max_samples = 5000  # Adjust as needed
if len(x_train) > max_samples:
    idx = np.random.choice(len(x_train), size=max_samples, replace=False)
    x_train_sub = x_train[idx]
    y_train_sub = y_train[idx]
else:
    x_train_sub = x_train
    y_train_sub = y_train

# ---------------------------
# Create a pipeline that transforms the input using RBF features and fits a Ridge regressor.
# ---------------------------
pipeline = Pipeline([
    ('rbf', RBFSampler(random_state=42)),
    ('ridge', Ridge(fit_intercept=True))
])

# ---------------------------
# Define a parameter distribution for RandomizedSearchCV
# ---------------------------
param_dist = {
    'rbf__n_components': [50, 100, 200, 400, 800],
    'rbf__gamma': [1e-4, 1e-3, 1e-2, 1e-1, 1],
    'ridge__alpha': [1e-4, 1e-3, 1e-2, 1e-1, 1, 10, 100]
}

# Use RandomizedSearchCV to search over the hyperparameters
random_search = RandomizedSearchCV(
    pipeline, 
    param_distributions=param_dist, 
    n_iter=20,      # try 20 random combinations
    cv=3,           # 3-fold cross-validation
    scoring='r2', 
    random_state=42, 
    n_jobs=-1
)
random_search.fit(x_train_sub, y_train_sub)

print("Best parameters found:")
print(random_search.best_params_)
print("Best cross-validation R^2: {:.3f}".format(random_search.best_score_))

# ---------------------------
# Evaluate the best model on the test set
# ---------------------------
y_pred = random_search.best_estimator_.predict(x_test)
test_r2 = r2_score(y_test, y_pred)
test_rmse = np.sqrt(mean_squared_error(y_test, y_pred))
print("\nTest set performance:")
print("Test R^2: {:.3f}".format(test_r2))
print("Test RMSE: {:.3f} °C".format(test_rmse))

# ---------------------------
# Visualize predictions vs. true values on the test set
# ---------------------------
plt.figure(figsize=(8, 6))
plt.scatter(x_test, y_test, color='blue', alpha=0.6, label='True Values')

import numpy as np
import matplotlib.pyplot as plt
from sklearn.kernel_approximation import RBFSampler
from sklearn.linear_model import Lasso
from sklearn.pipeline import Pipeline
from sklearn.model_selection import RandomizedSearchCV, train_test_split
from sklearn.metrics import mean_squared_error, r2_score

# ---------------------------
# Split data into training and test sets (80/20 split)
# ---------------------------
x_train, x_test, y_train, y_test = train_test_split(
    x, y, test_size=0.2, random_state=42
)

# ---------------------------
# Optionally subsample training data if it's very large to speed up hyperparameter search
# ---------------------------
max_samples = 5000  # Adjust this threshold as needed
if len(x_train) > max_samples:
    idx = np.random.choice(len(x_train), size=max_samples, replace=False)
    x_train_sub = x_train[idx]
    y_train_sub = y_train[idx]
else:
    x_train_sub = x_train
    y_train_sub = y_train

# ---------------------------
# Create a pipeline that transforms the input using RBF features and fits a Lasso regressor.
# ---------------------------
pipeline = Pipeline([
    ('rbf', RBFSampler(random_state=42)),
    ('lasso', Lasso(fit_intercept=True, max_iter=10000))
])

# ---------------------------
# Define a parameter distribution for RandomizedSearchCV
# ---------------------------
param_dist = {
    'rbf__n_components': [50, 100, 200, 400, 800],
    'rbf__gamma': [1e-4, 1e-3, 1e-2, 1e-1, 1],
    'lasso__alpha': [1e-4, 1e-3, 1e-2, 1e-1, 1, 10, 100]
}

# ---------------------------
# Use RandomizedSearchCV to search over the hyperparameters for the Lasso model
# ---------------------------
random_search = RandomizedSearchCV(
    pipeline,
    param_distributions=param_dist,
    n_iter=20,      # try 20 random combinations
    cv=3,           # 3-fold cross-validation
    scoring='r2', 
    random_state=42, 
    n_jobs=-1
)
random_search.fit(x_train_sub, y_train_sub)

print("Best parameters found:")
print(random_search.best_params_)
print("Best cross-validation R^2: {:.3f}".format(random_search.best_score_))

# ---------------------------
# Evaluate the best model on the test set
# ---------------------------
y_pred = random_search.best_estimator_.predict(x_test)
test_r2 = r2_score(y_test, y_pred)
test_rmse = np.sqrt(mean_squared_error(y_test, y_pred))
print("\nTest set performance:")
print("Test R^2: {:.3f}".format(test_r2))
print("Test RMSE: {:.3f} °C".format(test_rmse))
